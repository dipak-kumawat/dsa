#include<stdio.h>
void swap(int *x, int *y)
{
    int temp;
    temp=*x;
    *x=*y;
    *y=temp;
}

void quicksort(int a[20],int first, int last)
{
    int i,j,pivot;
    if(first<last)
    {
        pivot=a[first];
        i=first+1;
        j=last;
        while(i<=j)
        {
            while((i<=last) && (a[i]<=pivot))
                i++;
            while((j>=first) && (a[j]>pivot))
                j--;
            if(i<j)
            {
                swap(&a[i],&a[j]);
            }
        }
        swap(&a[first],&a[j]);
        quicksort(a,first,j-1);
        quicksort(a,j+1,last);
    }
}

int main()
{
    int a[20],i,n;
    printf("Enter size of array\n");
    scanf("%d",&n);
    printf("Enter the elements\n");
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    quicksort(a,0,n-1);
    printf("Sorted elements are\n");
    for(i=0;i<n;i++)
    {
        printf("%d\t",a[i]);
    }
    return 0;
}
